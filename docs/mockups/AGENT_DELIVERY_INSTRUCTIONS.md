### Phase 0b: Theme Provider Setup (Hour 2)
**Agent: Create React Context**

**File: `src/lib/theme-context.jsx`**
```jsx
'use client';

import { createContext, useContext, useState, useEffect } from 'react';
import { ThemeEngine } from './theme-engine';

const ThemeContext = createContext(null);

export function ThemeProvider({ children }) {
  const [engine] = useState(() => new ThemeEngine());
  const [state, setState] = useState({
    theme: engine.themeName,
    mode: engine.mode,
    customAccent: engine.customAccent,
  });

  useEffect(() => {
    // Apply theme on mount
    engine.applyTheme(document.documentElement);
    
    // Subscribe to changes
    const unsubscribe = engine.subscribe((newState) => {
      setState(newState);
    });
    
    return unsubscribe;
  }, [engine]);

  return (
    <ThemeContext.Provider value={{ engine, ...state }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
}
```

```jsx
export function Button({ children, variant = 'primary', ...props }) {
  const baseStyle = {
    padding: '0.5rem 1rem',
    borderRadius: '0.375rem',
    fontWeight: '500',
    cursor: 'pointer',
    border: 'none',
    transition: 'background-color 0.2s, opacity 0.2s',
    fontSize: 'inherit',
    fontFamily: 'inherit',
  };

  const variants = {
    primary: {
      ...baseStyle,
      backgroundColor: 'var(--primary)',
      color: 'var(--text-inverted)',
    },
    secondary: {
      ...baseStyle,
      backgroundColor: 'var(--bg-tertiary)',
      color: 'var(--text)',
      border: '1px solid var(--border)',
    },
    ghost: {
      ...baseStyle,
      backgroundColor: 'transparent',
      color: 'var(--primary)',
      border: '1px solid var(--primary)',
    },
  };

  return (
    <button
      style={variants[variant]}
      onMouseEnter={(e) => e.target.style.opacity = '0.8'}
      onMouseLeave={(e) => e.target.style.opacity = '1'}
      {...props}
    >
      {children}
    </button>
  );
}
```

## üö® Common Issues & Solutions

### Issue 1: "Theme not applying on page load"
**Cause:** `engine.applyTheme()` not called early enough

**Solution:**
```jsx
useEffect(() => {
  engine.applyTheme(document.documentElement); // Call FIRST
  const unsubscribe = engine.subscribe(...);
  return unsubscribe;
}, [engine]); // Run only once
```

### Issue 2: "Colors don't update when switching themes"
**Cause:** Components using hardcoded colors instead of CSS variables

**Solution:**
```jsx
// ‚ùå WRONG
<div style={{ backgroundColor: '#131316' }}>

// ‚úÖ CORRECT
<div style={{ backgroundColor: 'var(--bg-secondary)' }}>
```

### Issue 3: "Hydration warning in Next.js"
**Cause:** Server rendering doesn't know about client-side theme

**Solution:**
```jsx
// Add suppressHydrationWarning to <html>
<html suppressHydrationWarning>
```

### Issue 4: "localStorage not working"
**Cause:** Private browsing or storage disabled

**Solution:**
Already handled in theme-engine.js with try/catch fallback

### Issue 5: "Eyedropper not available"
**Cause:** Browser doesn't support EyeDropper API

**Solution:**
Already handled in color-picker.jsx - button hidden if unavailable

## üîí Security Considerations

### What's Safe
‚úÖ CSS variables (safe for user-generated colors)
‚úÖ localStorage (user's own browser only)

### What's Not Safe (don't do)
